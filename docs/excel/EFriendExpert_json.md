- codeInfo : 배열

  - name
  - zip
  - python
- TR_ID
  - info
    - name
    - version
    - method
    - productionDomain
    - developmentDomain
    - url
    - format
    - contentType
    - memo
    - trid
    - idProduct
    - description
  - request
    - header : field 배열
    - body : field 배열
  - response
    - header : field 배열
    - body : field 배열
  - field 배열
    - code
    - name
    - type
    - requireed
    - length
    - description
    - default
    - dnum
